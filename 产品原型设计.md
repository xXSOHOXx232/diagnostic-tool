### 1.3 目标用户
年GMV 500万-5000万的电商企业（淘宝/天猫/京东/拼多多）

## 四、评分逻辑设计

### 4.1 计算公式

```javascript
// 1. 维度得分 = Σ(题目得分 × 权重) / Σ权重
function calculateDimensionScore(dimension, answers) {
  let totalScore = 0;
  let totalWeight = 0;
  
  dimension.questions.forEach(question => {
    const answer = answers.find(a => a.questionId === question.id);
    if (answer) {
      totalScore += answer.score * question.weight;
      totalWeight += question.weight;
    }
  });
  
  return totalWeight > 0 ? totalScore / totalWeight : 0;
}

// 2. 总分 = Σ(维度得分) / 维度数 × 20
function calculateTotalScore(dimensionScores) {
  const avgScore = dimensionScores.reduce((sum, d) => sum + d.score, 0) / dimensionScores.length;
  return Math.round(avgScore * 20); // 转换为100分制
}

// 3. 健康度等级判定
function getHealthLevel(totalScore) {
  if (totalScore >= 85) return { level: "优秀", stars: 5 };
  if (totalScore >= 75) return { level: "良好", stars: 4 };
  if (totalScore >= 60) return { level: "一般", stars: 3 };
  if (totalScore >= 45) return { level: "较差", stars: 2 };
  return { level: "很差", stars: 1 };
}

// 4. 问题识别（得分<3.5的题目）
function identifyProblems(answers, questions) {
  const problems = [];
  
  answers.forEach(answer => {
    const question = questions.find(q => q.id === answer.questionId);
    if (answer.score < 3.5) {
      problems.push({
        severity: answer.score < 2.5 ? "high" : answer.score < 3.5 ? "medium" : "low",
        question: question.question,
        score: answer.score,
        suggestion: question.suggestion // 预设的改进建议
      });
    }
  });
  
  // 按严重程度排序
  return problems.sort((a, b) => {
    const severityOrder = { high: 0, medium: 1, low: 2 };
    return severityOrder[a.severity] - severityOrder[b.severity];
  });
}
```

// 计算超过百分比
function calculatePercentile(totalScore, distribution) {
  let lowerCount = 0;
  
  for (const range of distribution) {
    const [min, max] = range.range.split('-').map(Number);
    if (totalScore > max) {
      lowerCount += range.percentage;
    } else if (totalScore >= min && totalScore <= max) {
      // 线性插值
      const rangePercentage = (totalScore - min) / (max - min) * range.percentage;
      lowerCount += rangePercentage;
      break;
    }
  }
  
  return Math.round(lowerCount);
}
